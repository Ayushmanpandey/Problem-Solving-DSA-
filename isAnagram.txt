class Solution:
    
    #Function is to check whether two strings are anagram of each other or not.
    def isAnagram(self,a,b):
        #code here
        if(len(a)!=len(b)):
            return False
        else:
            for i in a:
                b= b.replace(i,"",1)
            if(len(b)==0):
                return True
            else:
                return False
